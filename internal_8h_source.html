266static inline int
267bio_vfmt(bio_fmt_buf_t* buf, const char* fmt, va_list args) {
268 va_list args_copy;
269 va_copy(args_copy, args);
270 int num_chars = vsnprintf(buf->ptr, (size_t)buf->len, fmt, args_copy);
271 va_end(args_copy);
272
273 if (num_chars < 0) { return num_chars; } // Format error
274
275 if (num_chars >= (int)buf->len) {
276 bio_free(buf->ptr);
277 buf->ptr = bio_malloc(num_chars + 1);
278 buf->len = num_chars + 1;
279
280 vsnprintf(buf->ptr, (size_t)buf->len, fmt, args);
281 }
282
283 buf->ptr[num_chars] = '\0';
284 return num_chars;
285}
